**理由：** 日线数据量少，可以用更短的回看周期。

#### 2. 添加震荡市检测

```python
def detect_range_bound_market(daily_data, lookback=60):
    """检测是否为箱体震荡"""
    # 1. 价格区间相对狭窄
    price_range = max(daily_data['close']) / min(daily_data['close']) - 1
    if price_range > 0.3:  # 波动超过30%
        return False

    # 2. 多次均值回归
    sma = daily_data['close'].rolling(20).mean()
    crosses = 0
    for i in range(1, len(daily_data)):
        if (daily_data['close'].iloc[i-1] < sma.iloc[i-1] and
            daily_data['close'].iloc[i] > sma.iloc[i]) or \
           (daily_data['close'].iloc[i-1] > sma.iloc[i-1] and
            daily_data['close'].iloc[i] < sma.iloc[i]):
            crosses += 1

    if crosses < 3:  # 回归次数太少
        return False

    return True  # 适合均值回归
```
